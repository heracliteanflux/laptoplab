def freq_to_bark (freq : float) -> float:
  bark = 6*np.arcsinh(freq/600)
  return bark

avg['F1_bark'] = avg['F1_Hz'].apply(freq_to_bark)
avg['F2_bark'] = avg['F2_Hz'].apply(freq_to_bark)
avg.drop(columns=['F1_Hz','F2_Hz']).to_csv('lab3_bark.csv')
